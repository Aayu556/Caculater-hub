<script>
loan.oninput = () => { loanVal.innerText = loan.value; calc(); }
rate.oninput = () => { rateVal.innerText = rate.value; calc(); }
time.oninput = () => { timeVal.innerText = time.value; calc(); }

function calc() {
  let P = loan.value;
  let r = rate.value / 100 / 12;
  let n = time.value * 12;

  let emi = (P*r*Math.pow(1+r,n)) / (Math.pow(1+r,n)-1);
  let total = emi * n;
  let interest = total - P;

  document.getElementById("emi").innerText = Math.round(emi);
  document.getElementById("principal").innerText = Math.round(P);
  document.getElementById("interest").innerText = Math.round(interest);
  document.getElementById("total").innerText = Math.round(total);
}

calc();
</script>
